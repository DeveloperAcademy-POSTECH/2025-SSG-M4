## Struct View

```swift
import SwiftUI

/// 좋아요버튼 Struct View로 분리
struct LikeButton: View {
    @Binding var isLiked: Bool
    let size: CGFloat = 24
    
    var body: some View {
        Button(action: { isLiked.toggle() }) {
            Image(systemName: isLiked ? "heart.fill" : "heart")
                .foregroundColor(isLiked ? .red : .gray)
                .font(.system(size: size))
        }
    }
}

/// 다른 View LikeButton그냥 땡겨와서 재사용이 된다. 
struct ContentView: View {
    @State private var liked = false
    
    var body: some View {
        VStack {
            Text("Struct로 만든 LikeButton")
            LikeButton(isLiked: $liked)
        }
    }
}

/// 다른 View에서도 뚝딱 재사용이 가능하다..우리가 사용하던 익숙한 방식 
struct ProductCard: View {
    @State private var isFavorite = false
    
    var body: some View {
        HStack {
            Text("상품명")
            Spacer()
            LikeButton(isLiked: $isFavorite)
        }
    }
}
```

**장점**: 여러 View에서 재사용 가능, 의존성 명확

## @ViewBuilder

```swift
import SwiftUI

struct ContentView: View {
    @State private var liked = false
    
    var body: some View {
        VStack {
            Text("@ViewBuilder 함수로 만든 LikeButton")
            likeButton(isLiked: $liked)
        }
    }
    
    /// @ViewBuilder 함수로 View를 반환
    @ViewBuilder
    private func likeButton(isLiked: Binding<Bool>) -> some View {
        Button(action: { isLiked.wrappedValue.toggle() }) {
            Image(systemName: isLiked.wrappedValue ? "heart.fill" : "heart")
                .foregroundColor(isLiked.wrappedValue ? .red : .gray)
        }
    }
}

/// 재사용해주려면 이걸 바로 못쓰고 추가 작업을 해야함 ! 
struct ProductCard: View {
    @State private var isFavorite = false
    
    var body: some View {
        HStack {
            Text("상품명")
            Spacer()
            // ContentView의 private 함수도 못쓰고
            // 그럼 함수를 public으로 바꾸던지
            // 코드 복붙하던지(이거할꺼면 나눈이유가없음..재사용못한거니까 )
            // extension으로 빼거나
            // 근데그럴거면 그냥 Struct로 새로만드는게 나을듯..!
        }
    }
}
```

**단점**: 재사용하려면 공개 범위 조정, extension 분리 필요

## Computed Property

```swift
import SwiftUI

struct ContentView: View {
    @State private var liked = false
    
    var body: some View {
        VStack {
            Text("Computed Property로 만든 LikeButton")
            likeButton
        }
    }
    
    /// computed 프로퍼티(간단하게 이해하면 변수로 빼줬다고 생각해도 될듯..!)
    var likeButton: some View {
        Button(action: { liked.toggle() }) {
            Image(systemName: liked ? "heart.fill" : "heart")
                .foregroundColor(liked ? .red : .gray)
        }
    }
}
```

**단점**: 파라미터 전달 불가, 일회성 코드에만 적합
